dd if = / dev / zero of = / var / swapfile bs = 1M count = 256＆amp; chmod 600 / var / swapfile＆amp;＆amp; mkswap / var / swapfile＆amp;＆amp; echo / var / swapfile none交换默认值0 0 | tee -a / etc / fstab＆amp;＆amp; swapon -a
